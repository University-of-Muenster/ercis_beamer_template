% Footline template (see beamer user guide, p. 61)
\makeatletter
\setbeamertemplate{footline}{%
  \ifbeamer@renderfootline
    {%
      \tikzset{external/export=false}%
      \begin{tikzpicture}
        % Picture size
        \useasboundingbox (0mm, \footlineHeight) rectangle (\paperwidth, 0mm);

		% Footline background
		\usebeamercolor{footline}%
		\path [fill, color={footline.bg}] (\leftTextMargin, 7.3mm) rectangle ({\paperwidth - \rightTextMargin}, 7mm);

		% Author
		\node[anchor=west, no sep] at (\leftTextMargin, {7mm - 2.25mm}) {%
		 \usebeamerfont{title page author}%
		 \usebeamercolor[fg]{title page author}%
		 \insertauthor \, | \insertshorttitle%
		};

        % Frame number
        \node[anchor=east, no sep] at ({\paperwidth - 5mm - 1.75mm}, {7mm - 2.25mm}) {%
          \usebeamerfont{frame number in foot}%
          \usebeamercolor[fg]{frame number in foot}%
          \insertframenumber%
        };

        % Navigation symbols (optional)
        % \ifbeamertemplateempty{navigation symbols}{}{%
        %   \node[anchor=west, no sep] at (7.5mm, 2mm) {%
        %     \usebeamertemplate***{navigation symbols}%
        %   };
        % }
      \end{tikzpicture}%
    }%
  \fi%
}
\makeatother
