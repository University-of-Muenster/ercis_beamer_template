% Define ERCIS logo for the headline
\newcommand{\headlineErcisLogo}{%
  \begin{tikzpicture}%
    \node [no sep] {%
      \resizebox{!}{12mm}{\ercislogo{ercisred}{ercisblue}{ercisred}}%
    };%
  \end{tikzpicture}%
}%

% Headline template (see beamer user guide, p. 59)
\makeatletter
\setbeamertemplate{headline}{%
  {%
    \tikzset{external/export=false}%
    \begin{tikzpicture}
      % Picture size
      \useasboundingbox (0mm, \headlineHeight) rectangle (\paperwidth, 0mm);

      % ERCIS logo
      \ifbeamer@renderheadlinelogo
        \node[anchor=north east, no sep] at ({\paperwidth - \rightTextMargin}, {\headlineHeight - 6mm}) {%
          \beamer@cacheUse{headline-ercis-logo}{\headlineErcisLogo}%
        };
      \fi

      % mininav
      \ifbeamer@renderheadlinenav
        \node[fill,color=white,anchor=north west, no sep, minimum width=\textwidth-\leftTextMargin-\rightTextMargin] at (\leftTextMargin, \headlineHeight-1mm) {%
          \insertsectionnavigationhorizontal{\linewidth}{}{}%
        };
      \fi
    \end{tikzpicture}%
  }%
}
\makeatother
